数的最小深度

 public int run(TreeNode root) {
    if(root==null){
        return 0;
    }
    int nleft=run(root.left);
    int nright=run(root.right);
    if(nleft==0||nright==0)
        return nleft+1+nright;  //这里要考虑斜树的状态
                若左子树为空，则返回右子树的最小深度+1；
                （加1是因为要加上根这一层，下同）
                若右子树为空，则返回左子树的最小深度+1；

    return Math.min(nleft,nright)+1;
}

按层遍历，第一个叶子节点就是最短路径

 public int run(TreeNode root) {
    if(root==null){
        return 0;
    }
    int level=1;
    TreeNode last=root;
    LinkedList<TreeNode> q= new LinkedList();
    int nowLine=1;int nextLine=0;
    q.add(root);
    while(!q.isEmpty()){
        TreeNode now=q.poll();
        nowLine--;
        if(now.left==null&&now.right==null)
            break;
        if(now.left!=null){
            q.add(now.left);
            nextLine++;
        }
        if(now.right!=null){
            q.add(now.right);
            nextLine++;
        }
        if(nowLine==0){
            level++;
            nowLine=nextLine;
            nextLine=0;
        }
    }
    return level;
}

public ListNode FindKthToTail(ListNode head,int k) {

   }
   return p2;

俩个合并的链表

输入两个单调递增的链表，输出两个链表合成后的链表，当然我们需要合成后的链表满足单调不减规则。

递归：每次都是表一和表2的值相比
public ListNode Merge(ListNode list1,ListNode list2) {
   if(list1 == null){
       return list2;//结束条件，俩个表有一个为空了
   }
   if(list2 == null){
       return list1;
   }
   if(list1.val <= list2.val){
       list1.next = Merge(list1.next, list2);
       return list1;//每次list.next不是在表一就是在表二，所以可以递归实现在
                       在每下一次递归时决定list.next是哪一个值
   }else{
       list2.next = Merge(list1, list2.next);
       return list2;
   }

}

复杂链表的复制

 public RandomListNode Clone(RandomListNode pHead)
{
    if(pHead==null)
        return null;
    RandomListNode pcur=pHead;
//第一步：复制next节点，如原来是A->B->C 变成A->A'->B->B'->C->C'
    while(pcur!=null){
        RandomListNode temp=new RandomListNode(pcur.label);新建复制节点
        temp.next=pcur.next;   
        pcur.next=temp;     //将复制节点插入到原链表中
        pcur=temp.next;     //移动节点继续复制
    }
    pcur=pHead;       //重新指向头结点
//第二步：复制随机节点
    while(pcur!=null){
        if(pcur.random!=null){
            pcur.next.random=pcur.random.next;  //随机节点复制（画图演示）
        }
        pcur=pcur.next.next;
    }
//第三部：链表的拆分
    pcur=pHead;   //重新指向头结点
    RandomListNode cloneHead=pHead.next;//用来返回的新节点头指针
    RandomListNode cur=cloneHead;  //用来操作的可变换的节点指针
    while(pcur!=null){
    //A->A'->B->B'->C->C'，所以拆分的时候隔俩个
        pcur.next=pcur.next.next;  //
        if(cur.next!=null)
            cur.next=cur.next.next;
        pcur=pcur.next;
        cur=cur.next;
    }
    return cloneHead;
}

反转链表

 public ListNode ReverseList(ListNode head) {
       ListNode p=head;  //记录当前节点
       ListNode result=null;//记录最终返回节点，用来指向最后一个节点
       ListNode pre=null;  //用来记录每次断开后的前一个节点
       while(p!=null){
            ListNode next=p.next;  //用来记录下一个节点，因为反转后链表断开了
            if(next==null)
                result=p;      //后节点为空时，说明是最后一个节点，可以在最后返回
            p.next=pre;      //当前节点的下一节点应该是前一个节点
            pre=p;           //更新下一次准备的前节点
            p=next;          //更新当前节点为之前记录的next节点
    }
    return result;
}

链表的环及入口

判断是否有环并返回相遇节点。
 public ListNode isLoop(ListNode pHead)
{
    if(pHead==null||pHead.next==null)
        return null;
     ListNode p1=pHead;
     ListNode p2=pHead;
     while(p1!=null&&p2!=null){
         p1=p1.next;
         p2=p2.next.next;
         if(p1==p2)     //有环迟早会相遇  
             return p2;
     }
     return null;
}
//找到入口节点
public ListNode EntryNodeOfLoop(ListNode pHead){
    ListNode mNode=isLoop(pHead);//先找到相遇节点
    ListNode p1=mNode;         
    if(p1==null)
        return null;
    ListNode p2=pHead;         //在设置一个指针指向头结点
    while(p1!=p2){              //开始同步走，再次相遇就是入口节点
        p1=p1.next;
        p2=p2.next;
    }
    return p1;
}

俩个链表的第一个公共节点

第一步先计算俩个链表的长度，然后长度长的链表节点先走(len1-len2)步，此时俩个节点距离相
交节点距离相同。然后开始一起走，直到遇到相同节点
 public ListNode FindFirstCommonNode(ListNode pHead1, ListNode pHead2) {
 if(pHead1==null||pHead2==null){
         return null;
     }
     int len1=1,len2=1;
     1.计算长度
     ListNode tail1=pHead1;
     while(tail1.next!=null){
         tail1=tail1.next;
         len1++;
     }
     ListNode tail2=pHead2;
     while(tail2.next!=null){
         tail2=tail2.next;
         len2++;
     }
     if(tail1!=tail2){
         return null;
     }
     2.快的先走len1-len步
     ListNode t1=pHead1,t2=pHead2;
     if(len1>len2){
         int d=len1-len2;
         while(d!=0){
             t1=t1.next;
             d--;
         }
     }else{
         int d=len2-len1;
         while(d!=0){
             t2=t2.next;
             d--;
         }
     }
    3.起步相同然后一起走
     while(t1!=t2){
         t1=t1.next;
         t2=t2.next;
     }
    return t1;
    }

保留原始节点去重：
public void delete(Node head){
    Node p=head;
    while(p!=null){    //外循环，从头到尾
        Node q=p;
        while(q.next!=null){//内循环，每次遍历当前节点之后所有节点，有重复去除
            if(p.data==q.next.data)
                q.next=q.next.next;
            else
                q=q.next;
        }
        p=p.next;
    }
}

对一个排序链表去重，不保留重复节点链表1->2->3->3->4->4->5 处理后为 1->2->5
 public ListNode deleteDuplication(ListNode pHead)
{
    ListNode first=new ListNode(-1);  //防止头结点丢失
    first.next=pHead;         
    ListNode cur=first;     //用来记录当前确认没有重复的节点，如上面的1，2，5
    ListNode p=pHead;       //用来遍历每一个节点，有重复就跳过
    while(p!=null&&p.next!=null){
        if(p.val==p.next.val){
            int val=p.val;   
            while(p!=null&&p.val==val) //有重复节点跳过
                p=p.next;
            cur.next=p;              //连接到新的数值，如2跳过3连到4
        }else{
            cur=cur.next;           //前后节点不重复，cur可以后移位置
            p=p.next;               p每次都要后移判断，直到遍历完成
        }
    }
     return first.next;      
}

插入排序

public ListNode insertionSortList(ListNode head) {

ListNode dumy = new ListNode(Integer.MIN_VALUE);//创建新链表
ListNode cur = head;     
ListNode pre = dumy;
while (cur != null) {
    //保存当前节点下一个节点
    ListNode next = cur.next;
    pre = dumy;//每次都从最开始的起点遍历
    //寻找当前节点正确位置的一个节点
    while (pre.next != null && pre.next.val < cur.val) {
        pre = pre.next;
    }
    //将当前节点加入新链表中
    cur.next = pre.next;
    pre.next = cur;
    //处理下一个节点
    cur = next;
}
return dumy.next;
